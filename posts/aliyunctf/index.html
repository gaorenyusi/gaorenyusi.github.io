import sys  
import math  
import collections  
import queue  
import heapq  
import bisect  
  
def audit_checker(event,args):  
    if not event in ["import","time.sleep","builtins.input","builtins.input/result"]:        
    	raise RuntimeError  
sys.addaudithook(audit_checker)
